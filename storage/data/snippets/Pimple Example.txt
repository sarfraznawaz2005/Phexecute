use Pimple\Container;

class Foo {
   public function bar() {
      return "Iam bar!";
   }
}

$container = new Container();

// define our services
$container['foo'] = function ($c) {
    return new Foo;
};

$foo = $container['foo'];
echo $foo->bar();
